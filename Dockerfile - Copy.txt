# Use an official Python runtime as a parent image
FROM python:3.9-slim

# Set the working directory in the container
WORKDIR /app

# Copy the current directory contents into the container at /app
COPY . /app

# Install any needed packages specified in requirements.txt
COPY requirements.txt ./
RUN pip install --no-cache-dir -r requirements.txt

# Install Mistral 8x22b and the tokenizer
RUN python -c "from transformers import AutoModelForCausalLM, AutoTokenizer; model_name='mistralai/Mixtral-8x22B-v0.1'; model = AutoModelForCausalLM.from_pretrained(model_name); tokenizer = AutoTokenizer.from_pretrained(model_name); model.save_pretrained('/app/model'); tokenizer.save_pretrained('/app/model')"
